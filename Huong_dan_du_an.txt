CREATE DATABASE QL_GYM
GO
USE QL_GYM
GO
CREATE TABLE TaiKhoan (
    Username NVARCHAR(50) NOT NULL,
    Password NVARCHAR(50) NOT NULL,
    Role NVARCHAR(50) NOT NULL,
    CONSTRAINT PK_TaiKhoan PRIMARY KEY (Username)
)
CREATE TABLE HoiVien (
    MaHV INT IDENTITY(1,1) NOT NULL,
    HoTen NVARCHAR(100) NOT NULL,
    NgaySinh DATE,
    GioiTinh NVARCHAR(10),
    SDT NVARCHAR(15),
    CONSTRAINT PK_HoiVien PRIMARY KEY (MaHV)
)
CREATE TABLE GoiTap (
    MaGoi INT IDENTITY(1,1) NOT NULL,
    TenGoi NVARCHAR(100) NOT NULL,
    Gia MONEY NOT NULL,
    ThoiHan INT NOT NULL, -- số ngày
    CONSTRAINT PK_GoiTap PRIMARY KEY (MaGoi)
)
CREATE TABLE HLV (
    MaHLV INT IDENTITY(1,1) NOT NULL,
    HoTen NVARCHAR(100) NOT NULL,
    ChuyenMon NVARCHAR(100),
    SDT NVARCHAR(15),
    CONSTRAINT PK_HLV PRIMARY KEY (MaHLV)
)
CREATE TABLE DangKy (
    MaDK INT IDENTITY(1,1) NOT NULL,
    MaHV INT NOT NULL,
    MaGoi INT NOT NULL,
    MaHLV INT NOT NULL,
    NgayDK DATE NOT NULL,
    NgayHetHan DATE NOT NULL,
    CONSTRAINT PK_DangKy PRIMARY KEY (MaDK),
    CONSTRAINT FK_DangKy_HoiVien FOREIGN KEY (MaHV) REFERENCES HoiVien(MaHV),
    CONSTRAINT FK_DangKy_GoiTap FOREIGN KEY (MaGoi) REFERENCES GoiTap(MaGoi),
    CONSTRAINT FK_DangKy_HLV FOREIGN KEY (MaHLV) REFERENCES HLV(MaHLV)
)
CREATE TABLE HoaDon (
    MaHD INT IDENTITY(1,1) NOT NULL,
    MaDK INT NOT NULL,
    NgayLap DATE NOT NULL,
    TongTien MONEY NOT NULL,
    CONSTRAINT PK_HoaDon PRIMARY KEY (MaHD),
    CONSTRAINT FK_HoaDon_DangKy FOREIGN KEY (MaDK) REFERENCES DangKy(MaDK)
)
